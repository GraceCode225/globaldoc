TABLE AVIS CLIENTS ── -- Permet aux visiteurs de laisser un commentaire et une note CREATE TABLE IF NOT EXISTS avis_clients ( id BIGSERIAL PRIMARY KEY, nom TEXT NOT NULL, service TEXT DEFAULT 'Service GlobalDoc', texte TEXT NOT NULL, etoiles INTEGER DEFAULT 5 CHECK (etoiles BETWEEN 1 AND 5), created_at TIMESTAMPTZ DEFAULT NOW() ); -- Sécurité RLS ALTER TABLE avis_clients ENABLE ROW LEVEL SECURITY; -- Tout le monde peut lire les avis DROP POLICY IF EXISTS "avis_read" ON avis_clients; CREATE POLICY "avis_read" ON avis_clients FOR SELECT USING (true); -- Tout le monde peut laisser un avis (INSERT) DROP POLICY IF EXISTS "avis_insert" ON avis_clients; CREATE POLICY "avis_insert" ON avis_clients FOR INSERT WITH CHECK (true); -- Seul l'admin connecté peut supprimer DROP POLICY IF EXISTS "avis_delete" ON avis_clients; CREATE POLICY "avis_delete" ON avis_clients FOR DELETE USING (auth.role() = 'authenticated'); -- ── 2. TABLE CONFIG (contrôle du site depuis l'admin) ── CREATE TABLE IF NOT EXISTS config ( id BIGSERIAL PRIMARY KEY, cle TEXT UNIQUE NOT NULL, valeur TEXT DEFAULT '', updated_at TIMESTAMPTZ DEFAULT NOW() ); -- Valeurs par défaut INSERT INTO config (cle, valeur) VALUES ('bibliotheque', 'true'), ('flashsale', 'true'), ('communaute', 'true'), ('commandes', 'true'), ('whatsapp', '2250142116172'), ('wave', '0501233818'), ('promo_text', '🔥 Offre spéciale : CV + Lettre à 1300F seulement !'), ('promo_date', '') ON CONFLICT (cle) DO NOTHING; -- Sécurité RLS ALTER TABLE config ENABLE ROW LEVEL SECURITY; DROP POLICY IF EXISTS "config_read" ON config; CREATE POLICY "config_read" ON config FOR SELECT USING (true); DROP POLICY IF EXISTS "config_write" ON config; CREATE POLICY "config_write" ON config FOR ALL USING (auth.role() = 'authenticated'); -- ── VÉRIFICATION ── SELECT table_name, 'OK ✅' AS status FROM information_schema.tables WHERE table_schema = 'public' AND table_name IN ('avis_clients', 'config') ORDER BY table_name; -- ── 3. TABLE MEDIAS (photos affiches + vidéos) ── CREATE TABLE IF NOT EXISTS medias ( id BIGSERIAL PRIMARY KEY, type TEXT NOT NULL CHECK (type IN ('photo','video')), titre TEXT NOT NULL, sous_titre TEXT DEFAULT '', image_url TEXT, video_url TEXT, lien TEXT DEFAULT 'commander.html', bouton TEXT DEFAULT 'Voir →', created_at TIMESTAMPTZ DEFAULT NOW() ); ALTER TABLE medias ENABLE ROW LEVEL SECURITY; DROP POLICY IF EXISTS "medias_read" ON medias; DROP POLICY IF EXISTS "medias_insert" ON medias; DROP POLICY IF EXISTS "medias_delete" ON medias; CREATE POLICY "medias_read" ON medias FOR SELECT USING (true); CREATE POLICY "medias_insert" ON medias FOR INSERT WITH CHECK (auth.role() = 'authenticated'); CREATE POLICY "medias_delete" ON medias FOR DELETE USING (auth.role() = 'authenticated'); -- ── BUCKET STORAGE "medias" (pour les photos affiches) ── INSERT INTO storage.buckets (id, name, public, file_size_limit, allowed_mime_types) VALUES ( 'medias', 'medias', true, 10485760, ARRAY['image/jpeg','image/png','image/webp','image/gif'] ) ON CONFLICT (id) DO UPDATE SET public = true; DROP POLICY IF EXISTS "medias_stor_read" ON storage.objects; DROP POLICY IF EXISTS "medias_stor_insert" ON storage.objects; DROP POLICY IF EXISTS "medias_stor_delete" ON storage.objects; CREATE POLICY "medias_stor_read" ON storage.objects FOR SELECT USING (bucket_id = 'medias'); CREATE POLICY "medias_stor_insert" ON storage.objects FOR INSERT WITH CHECK (bucket_id = 'medias'); CREATE POLICY "medias_stor_delete" ON storage.objects FOR DELETE USING (bucket_id = 'medias');
